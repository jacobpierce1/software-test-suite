import matplotlib.pyplot as plt 
import numpy as np 

x = np.arange(10)
y = np.copy( x )

def make_plot( title ) : 
    ax = plt.axes()
    ax.set_title( title, fontsize = 20 ) 
    ax.set_xlabel( 'X axis', fontsize = 16 )
    ax.set_ylabel( 'Y axis', fontsize = 16 )
    ax.plot( x, y ) 
    plt.show()
